# BibleClient

The `BibleClient` class provides methods for fetching Bible data from the YouVersion Platform API. It supports retrieving Bible versions, books, chapters, verses, passages, and the Verse of the Day (VOTD).

## Constructor

```typescript
import { ApiClient, BibleClient } from '@youversion/platform-core';

const apiClient = new ApiClient({ appKey: 'your-app-key' });
const bibleClient = new BibleClient(apiClient);
```

<ParamField path="client" type="ApiClient" required>
  An instance of `ApiClient` used to make HTTP requests.
</ParamField>

## Methods

### getVersions()

Fetches a collection of Bible versions filtered by language ranges.

```typescript
async getVersions(
  language_ranges: string | string[],
  license_id?: string | number,
  options?: GetVersionsOptions
): Promise<Collection<BibleVersion>>
```

<ParamField path="language_ranges" type="string | string[]" required>
  One or more language codes or ranges to filter versions (e.g., `"en"`, `"en*"`, `["en", "es"]`).
  
  Supports wildcards: `"en*"` matches all English variants.
</ParamField>

<ParamField path="license_id" type="string | number">
  Optional license ID to filter versions by specific licensing.
</ParamField>

<ParamField path="options" type="GetVersionsOptions">
  Optional pagination and field selection options.
  
  <Expandable title="GetVersionsOptions properties">
    <ParamField path="page_size" type="number | '*'">
      Number of results per page, or `"*"` to return all results (requires 1-3 fields).
    </ParamField>
    
    <ParamField path="page_token" type="string">
      Token for fetching the next page of results.
    </ParamField>
    
    <ParamField path="fields" type="Array<keyof BibleVersion>">
      Specific fields to return (e.g., `['id', 'title', 'language_tag']`).
    </ParamField>
    
    <ParamField path="all_available" type="boolean">
      Include all available versions, not just active ones.
    </ParamField>
  </Expandable>
</ParamField>

<ResponseField name="Returns" type="Promise<Collection<BibleVersion>>">
  A collection of Bible versions matching the language criteria.
  
  <Expandable title="BibleVersion properties">
    <ResponseField name="id" type="number">
      Bible version identifier
    </ResponseField>
    <ResponseField name="abbreviation" type="string">
      Version abbreviation (e.g., "NIV", "KJV")
    </ResponseField>
    <ResponseField name="title" type="string">
      Full version title
    </ResponseField>
    <ResponseField name="localized_title" type="string">
      Title in the version's language
    </ResponseField>
    <ResponseField name="localized_abbreviation" type="string">
      Abbreviation in the version's language
    </ResponseField>
    <ResponseField name="language_tag" type="string">
      BCP 47 language code (e.g., "en", "es")
    </ResponseField>
    <ResponseField name="books" type="string[]">
      Array of book identifiers in this version (e.g., ["GEN", "EXO"])
    </ResponseField>
    <ResponseField name="youversion_deep_link" type="string">
      Deep link URL to YouVersion app
    </ResponseField>
    <ResponseField name="copyright" type="string | null">
      Short copyright text
    </ResponseField>
    <ResponseField name="promotional_content" type="string | null">
      Long copyright text
    </ResponseField>
    <ResponseField name="info" type="string | null">
      Bible information text
    </ResponseField>
    <ResponseField name="publisher_url" type="string | null">
      Publisher's website URL
    </ResponseField>
    <ResponseField name="organization_id" type="string | null">
      Publisher organization ID
    </ResponseField>
  </Expandable>
</ResponseField>

#### Examples

<CodeGroup>
```typescript Basic Usage
// Get all English Bible versions
const versions = await bibleClient.getVersions('en*');

console.log(`Found ${versions.data.length} versions`);
versions.data.forEach(version => {
  console.log(`${version.abbreviation}: ${version.title}`);
});
```

```typescript Multiple Languages
// Get versions in English and Spanish
const versions = await bibleClient.getVersions(['en', 'es']);

// Filter by language
const englishVersions = versions.data.filter(v => v.language_tag === 'en');
const spanishVersions = versions.data.filter(v => v.language_tag === 'es');
```

```typescript With Pagination
// Get first page
const page1 = await bibleClient.getVersions('en*', undefined, {
  page_size: 10
});

// Get next page if available
if (page1.next_page_token) {
  const page2 = await bibleClient.getVersions('en*', undefined, {
    page_size: 10,
    page_token: page1.next_page_token
  });
}
```

```typescript Specific Fields
// Only fetch specific fields for better performance
const versions = await bibleClient.getVersions('en*', undefined, {
  page_size: '*',
  fields: ['id', 'title', 'abbreviation']
});
```
</CodeGroup>

### getBooks()

Fetches all books for a given Bible version.

```typescript
async getBooks(
  versionId: number,
  canon?: CANON
): Promise<Collection<BibleBook>>
```

<ParamField path="versionId" type="number" required>
  The Bible version ID.
</ParamField>

<ParamField path="canon" type="CANON">
  Optional canon filter: `"old_testament"`, `"new_testament"`, or `"deuterocanon"`.
</ParamField>

<ResponseField name="Returns" type="Promise<Collection<BibleBook>>">
  Collection of Bible books. Each book may include an optional `intro` field with metadata for book introductions.
  
  <Expandable title="BibleBook properties">
    <ResponseField name="id" type="string">
      Book identifier (e.g., "GEN", "MAT")
    </ResponseField>
    <ResponseField name="title" type="string">
      Book title (e.g., "Genesis")
    </ResponseField>
    <ResponseField name="full_title" type="string">
      Full book title (e.g., "The First Book of Moses, Commonly Called Genesis")
    </ResponseField>
    <ResponseField name="abbreviation" type="string">
      Book abbreviation (e.g., "Gen")
    </ResponseField>
    <ResponseField name="canon" type="CANON">
      Canonical section: `"old_testament"`, `"new_testament"`, or `"deuterocanon"`
    </ResponseField>
    <ResponseField name="intro" type="BibleBookIntro">
      Optional book introduction metadata
      - `id`: Intro identifier
      - `passage_id`: Passage ID for the intro content
      - `title`: Intro title
    </ResponseField>
  </Expandable>
</ResponseField>

#### Examples

<CodeGroup>
```typescript All Books
// Get all books in a version
const books = await bibleClient.getBooks(111);

books.data.forEach(book => {
  console.log(`${book.id}: ${book.title} (${book.canon})`);
});
```

```typescript Filter by Canon
// Get only Old Testament books
const otBooks = await bibleClient.getBooks(111, 'old_testament');

// Get only New Testament books
const ntBooks = await bibleClient.getBooks(111, 'new_testament');

// Get only Deuterocanonical books
const dcBooks = await bibleClient.getBooks(111, 'deuterocanon');
```

```typescript With Introductions
// Check for book introductions
const books = await bibleClient.getBooks(111);

const booksWithIntros = books.data.filter(book => book.intro);
for (const book of booksWithIntros) {
  console.log(`${book.title} has an introduction`);
  
  // Fetch the intro content
  if (book.intro) {
    const intro = await bibleClient.getPassage(111, book.intro.passage_id);
    console.log(intro.content);
  }
}
```
</CodeGroup>

### getBook()

Fetches a specific book by USFM code for a given version.

```typescript
async getBook(versionId: number, book: string): Promise<BibleBook>
```

<ParamField path="versionId" type="number" required>
  The Bible version ID.
</ParamField>

<ParamField path="book" type="string" required>
  The 3-character book identifier code (e.g., "GEN", "MAT", "REV").
</ParamField>

<ResponseField name="Returns" type="Promise<BibleBook>">
  The requested Bible book, which may include an optional `intro` field. Use the `passage_id` from the intro with `getPassage()` to fetch introduction content.
</ResponseField>

#### Example

```typescript
// Get the book of John
const john = await bibleClient.getBook(111, 'JHN');

console.log(john.title); // "John"
console.log(john.full_title); // "The Gospel According to John"
console.log(john.canon); // "new_testament"

// Check for introduction
if (john.intro) {
  const intro = await bibleClient.getPassage(111, john.intro.passage_id);
  console.log(intro.content);
}
```

### getPassage()

Fetches a passage (range of verses) from the Bible using the passages endpoint. Returns HTML or text formatted content.

```typescript
async getPassage(
  versionId: number,
  usfm: string,
  format?: 'html' | 'text',
  include_headings?: boolean,
  include_notes?: boolean
): Promise<BiblePassage>
```

<ParamField path="versionId" type="number" required>
  The Bible version ID.
</ParamField>

<ParamField path="usfm" type="string" required>
  The USFM reference. Supports:
  - Single verse: `"JHN.3.16"`
  - Verse range: `"JHN.3.1-5"`
  - Entire chapter: `"GEN.1"`
  - Multiple chapters: `"GEN.1-2"`
</ParamField>

<ParamField path="format" type="'html' | 'text'" default="html">
  The format to return. Use `"html"` for formatted content or `"text"` for plain text.
</ParamField>

<ParamField path="include_headings" type="boolean">
  Whether to include section headings in the content.
</ParamField>

<ParamField path="include_notes" type="boolean">
  Whether to include footnotes and study notes in the content.
</ParamField>

<ResponseField name="Returns" type="Promise<BiblePassage>">
  The requested passage with formatted content.
  
  <Expandable title="BiblePassage properties">
    <ResponseField name="id" type="string">
      Passage identifier (e.g., "JHN.3.16")
    </ResponseField>
    <ResponseField name="content" type="string">
      Passage content (HTML or plain text based on format parameter)
    </ResponseField>
    <ResponseField name="reference" type="string">
      Human-readable reference (e.g., "John 3:16")
    </ResponseField>
  </Expandable>
</ResponseField>

#### Examples

<CodeGroup>
```typescript Single Verse
// Get John 3:16
const verse = await bibleClient.getPassage(111, 'JHN.3.16');

console.log(verse.reference); // "John 3:16"
console.log(verse.content); // HTML content
```

```typescript Verse Range
// Get John 3:1-5
const passage = await bibleClient.getPassage(111, 'JHN.3.1-5');

console.log(passage.reference); // "John 3:1-5"
```

```typescript Entire Chapter
// Get Genesis 1

console.log(chapter.reference); // "Genesis 1"
```

```typescript Plain Text
// Get passage as plain text instead of HTML
const passage = await bibleClient.getPassage(
  111,
  'JHN.3.16',
  'text'
);

console.log(passage.content); // Plain text without HTML tags
```

```typescript With Headings and Notes
// Include section headings and footnotes

  111,
  'JHN.3.1-21',
  'html',
  true,  // include headings
  true   // include notes
);
```

### getIndex()

Fetches the complete indexing structure for a Bible version, containing the full hierarchy of books, chapters, and verses.

```typescript
async getIndex(versionId: number): Promise<BibleIndex>
```

<ParamField path="versionId" type="number" required>
  The Bible version ID.
</ParamField>

<ResponseField name="Returns" type="Promise<BibleIndex>">
  The complete Bible index structure.
  
  <Expandable title="BibleIndex structure">
    <ResponseField name="text_direction" type="string">
      Text direction for the version (e.g., "ltr", "rtl")
    </ResponseField>
    <ResponseField name="books" type="BibleIndexBook[]">
      Array of books with complete chapter and verse structures
      
      Each book contains:
      - `id`: Book identifier
      - `title`: Book title
      - `full_title`: Full book title
      - `abbreviation`: Book abbreviation
      - `canon`: Canonical section
      - `chapters`: Array of chapters with verses
    </ResponseField>
  </Expandable>
</ResponseField>

#### Example

```typescript
// Get the complete index for NIV
const index = await bibleClient.getIndex(111);

console.log(`Text direction: ${index.text_direction}`);
console.log(`Total books: ${index.books.length}`);

// Navigate the structure
index.books.forEach(book => {
  console.log(`\n${book.title} (${book.id})`);
  console.log(`  Chapters: ${book.chapters.length}`);
  
  book.chapters.forEach(chapter => {
    console.log(`  Chapter ${chapter.title}: ${chapter.verses.length} verses`);
  });
});

// Find a specific verse
const genesis = index.books.find(b => b.id === 'GEN');
const chapter1 = genesis?.chapters.find(c => c.title === '1');
const verse1 = chapter1?.verses.find(v => v.title === '1');
console.log(verse1?.id); // Verse ID
```

    </ResponseField>
    <ResponseField name="passage_id" type="string">

#### Example

```typescript

console.log(`Total VOTD entries: ${votds.data.length}`);

const dayOfYear = Math.floor(
  (today - new Date(today.getFullYear(), 0, 0)) / 86400000
);

if (todayVOTD) {
  const passage = await bibleClient.getPassage(111, todayVOTD.passage_id);
  console.log(`Today's verse: ${passage.reference}`);
  console.log(passage.content);
```

## Complete Example

```typescript
import { ApiClient, BibleClient } from '@youversion/platform-core';

// Initialize clients
const apiClient = new ApiClient({
  appKey: process.env.YOUVERSION_APP_KEY!,
});
const bibleClient = new BibleClient(apiClient);

// Find English versions
const versions = await bibleClient.getVersions('en*');
const niv = versions.data.find(v => v.abbreviation === 'NIV');

if (niv) {
  console.log(`Using ${niv.title}`);
  
  // Get the books
  const books = await bibleClient.getBooks(niv.id);
  console.log(`${books.data.length} books available`);
  
  // Get New Testament only
  const ntBooks = await bibleClient.getBooks(niv.id, 'new_testament');
  console.log(`${ntBooks.data.length} NT books`);
  
  // Get John 3:16
  const verse = await bibleClient.getPassage(niv.id, 'JHN.3.16');
  console.log(`\n${verse.reference}`);
  console.log(verse.content);
  
  // Get today's verse of the day
  const today = new Date();
  const dayOfYear = Math.floor(
    (today - new Date(today.getFullYear(), 0, 0)) / 86400000
  );
  const votd = await bibleClient.getVOTD(dayOfYear);
  const votdPassage = await bibleClient.getPassage(niv.id, votd.passage_id);
  
  console.log(`\nVerse of the Day: ${votdPassage.reference}`);
  console.log(votdPassage.content);
}
```

## Best Practices

<AccordionGroup>
  <Accordion title="Use wildcards for language matching">
    When searching for versions, use wildcards to match all language variants:
    
    ```typescript
    // Match all English variants (en, en-US, en-GB, etc.)
    const versions = await bibleClient.getVersions('en*');
    
    // Match specific language
    const spanishVersions = await bibleClient.getVersions('es');
    ```
  </Accordion>
  
  <Accordion title="Fetch passages instead of individual verses">
    For better performance, use `getPassage()` to fetch multiple verses at once:
    
    ```typescript
    // ❌ Not recommended - multiple API calls
    for (let i = 1; i <= 5; i++) {
      const verse = await bibleClient.getVerse(111, 'JHN', 3, i);
      // ...
    }
    
    // ✅ Recommended - single API call
    const passage = await bibleClient.getPassage(111, 'JHN.3.1-5');
    ```
  </Accordion>
  
  <Accordion title="Cache version and book data">
    Version and book metadata rarely changes, so cache it to reduce API calls:
    
    ```typescript
    // Cache versions
    const versionsCache = new Map<string, Collection<BibleVersion>>();
    
    async function getVersionsCached(lang: string) {
      if (!versionsCache.has(lang)) {
        const versions = await bibleClient.getVersions(lang);
        versionsCache.set(lang, versions);
      }
      return versionsCache.get(lang)!;
    }
    ```
  </Accordion>
  
  <Accordion title="Handle pagination for large result sets">
    When fetching all results, handle pagination properly:
    
    ```typescript
    async function getAllVersions(language: string) {
      const allVersions: BibleVersion[] = [];
      let nextToken: string | null = null;
      
      do {
        const result = await bibleClient.getVersions(language, undefined, {
          page_size: 50,
          page_token: nextToken || undefined
        });
        
        allVersions.push(...result.data);
        nextToken = result.next_page_token;
      } while (nextToken);
      
      return allVersions;
    }
    ```
  </Accordion>
  
  <Accordion title="Validate user input for USFM references">
    When accepting user input for passages, validate the format:
    
    ```typescript
    function isValidUSFM(usfm: string): boolean {
      // Basic validation for USFM format
      return /^[A-Z]{3}\.\d+(\.\d+)?(-\d+)?$/.test(usfm);
    }
    
    const userInput = 'JHN.3.16';
    if (isValidUSFM(userInput)) {
      const passage = await bibleClient.getPassage(111, userInput);
    } else {
      console.error('Invalid USFM format');
    }
    ```
  </Accordion>
</AccordionGroup>
